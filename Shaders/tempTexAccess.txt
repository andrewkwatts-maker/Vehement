	else
	{
		vec3 DeltaN = normalize(vPosition-LightPos);
		vec3 DeltaNA = abs(DeltaN);
		float Max = max(DeltaNA.x,max(DeltaNA.y,DeltaNA.z));

		vec3 MultDelt = DeltaN*(1/Max);

		if(DeltaNA.x == Max)
		{
			if(DeltaN.x > 0)
			{
				if(v<=texture(XP,MultDelt.yz/2)

				else

			}
			else

			else

			}
		}
		else
		{
			if(DeltaN.y > 0)
			{
				if(v<=texture(YP,MultDelt.xz)

				else
